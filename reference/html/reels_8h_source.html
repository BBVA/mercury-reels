491 double linear_correlation(OptimizeEval &ev) {
492 ExtFloat s_h = 0, s_o = 0, sho = 0, ssh = 0, sso = 0;
493 int n = 0;
494
495 for (OptimizeEval::iterator it = ev.begin(); it != ev.end(); ++it) {
496 if (it->t_obs != 0) {
497 s_h += it->t_hat;
498 s_o += it->t_obs;
499 sho += it->t_hat*it->t_obs;
500 ssh += it->t_hat*it->t_hat;
501 sso += it->t_obs*it->t_obs;
502
503 n++;
504 }
505 }
506 if (n == 0)
507 return 0;
508
509 double d2 = (n*ssh - s_h*s_h)*(n*sso - s_o*s_o);
510
511 if (d2 <= 1e-20)
512 return 0;
513
514 return (n*sho - s_h*s_o)/sqrt(d2);
515 }

1097 inline int update_node(int idx_parent, uint64_t code, bool target, ExtFloat time_d) {
1098
1099 if (idx_parent == 0) { // The root node contains the prediction of the zero-length clip.
1100 tree[0].n_seen++;
1101 if (target) {
1102 tree[0].n_target++;
1103 tree[0].sum_time_d += time_d;
1104 }
1105 }
1106
1107 ChildIndex::iterator it = tree[idx_parent].child.find(code);
1108
1109 if (it != tree[idx_parent].child.end()) {
1110 int idx = it->second;
1111
1112 tree[idx].n_seen++;
1113 if (target) {
1114 tree[idx].n_target++;
1115 tree[idx].sum_time_d += time_d;
1116 }
1117
1118 return idx;
1119 }
1120
1121 CodeTreeNode node = {1, target, time_d, {}};
1122
1123 tree.push_back(node);
1124
1125 int idx = tree.size() - 1;
1126
1127 tree[idx_parent].child[code] = idx;
1128
1129 return idx;
1130 }

1174 inline double agresti_coull_upper_bound(uint64_t n_hits, uint64_t n_total) {
1175 // https://github.com/msn0/agresti-coull-interval/blob/master/src/agresti.js
1176
1177 double n_tilde = n_total + binomial_z_sqr;
1178 double p_tilde = (n_hits + binomial_z_sqr_div_2)/n_tilde;
1179 double a = binomial_z*sqrt(p_tilde*(1 - p_tilde)/n_tilde);
1180
1181 return p_tilde + a;
1182 }

1195 inline double agresti_coull_lower_bound(uint64_t n_hits, uint64_t n_total) {
1196 // https://github.com/msn0/agresti-coull-interval/blob/master/src/agresti.js

1199 double p_tilde = (n_hits + binomial_z_sqr_div_2)/n_tilde;

1229 inline double predict_time(CodeTreeNode &node) {
1230
1231 if (node.n_target <= 0)
1232 return PREDICT_MAX_TIME;
1233
1234 double lb = std::max(1e-4, agresti_coull_lower_bound(node.n_target, node.n_seen));
1235 double mu_hat = transform == tr_linear ? ((double) node.sum_time_d)/node.n_target : exp(((double) node.sum_time_d)/node.n_target);
1236
1237 return mu_hat/lb;
1238 }

1247 inline double predict_clip(Clip clip) {
1248
1249 int idx = 0, n = 0;
1250
1251 double t[MAX_SEQ_LEN_IN_PREDICT];
1252
1253 for (Clip::reverse_iterator it = clip.rbegin(); it != clip.rend(); it++) {
1254 ChildIndex::iterator jt = tree[idx].child.find(it->second);
1255
1256 if (jt == tree[idx].child.end())
1257 break;
1258
1259 idx = jt->second;
1260
1261 t[n++] = predict_time(tree[idx]);
1262 }
1263
1264 if (n == 0)
1265 return predict_time(tree[0]);
1266
1267 if (aggregate == ag_longest)
1268 return t[n - 1];
1269
1270 double ret = t[0];
1271
1272 if (aggregate == ag_mean) {
1273 for (int i = 1; i < n; i++)
1274 ret += t[i];
1275
1276 return ret/n;
1277 }
1278
1279 for (int i = 1; i < n; i++)
1280 ret = std::min(ret, t[i]);
1281
1282 return ret;
1283 }
