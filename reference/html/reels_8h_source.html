491 double linear_correlation(OptimizeEval &ev) {
492 ExtFloat s_h = 0, s_o = 0, sho = 0, ssh = 0, sso = 0;
493 int n = 0;
494
495 for (OptimizeEval::iterator it = ev.begin(); it != ev.end(); ++it) {
496 if (it->t_obs != 0) {
497 s_h += it->t_hat;
498 s_o += it->t_obs;
499 sho += it->t_hat*it->t_obs;
500 ssh += it->t_hat*it->t_hat;
501 sso += it->t_obs*it->t_obs;
502
503 n++;
504 }
505 }
506 if (n == 0)
507 return 0;
508
509 double d2 = (n*ssh - s_h*s_h)*(n*sso - s_o*s_o);
510
511 if (d2 <= 1e-20)
512 return 0;
513
514 return (n*sho - s_h*s_o)/sqrt(d2);
515 }

1090 inline int update_node(int idx_parent, uint64_t code, bool target, ExtFloat time_d) {
1091
1092 if (idx_parent == 0) { // The root node contains the prediction of the zero-length clip.
1093 tree[0].n_seen++;
1094 if (target) {
1095 tree[0].n_target++;
1096 tree[0].sum_time_d += time_d;
1097 }
1098 }
1099
1100 ChildIndex::iterator it = tree[idx_parent].child.find(code);
1101
1102 if (it != tree[idx_parent].child.end()) {
1103 int idx = it->second;
1104
1105 tree[idx].n_seen++;
1106 if (target) {
1107 tree[idx].n_target++;
1108 tree[idx].sum_time_d += time_d;
1109 }
1110
1111 return idx;
1112 }
1113
1114 CodeTreeNode node = {1, target, time_d, {}};
1115
1116 tree.push_back(node);
1117
1118 int idx = tree.size() - 1;
1119
1120 tree[idx_parent].child[code] = idx;
1121
1122 return idx;
1123 }

1167 inline double agresti_coull_upper_bound(uint64_t n_hits, uint64_t n_total) {
1168 // https://github.com/msn0/agresti-coull-interval/blob/master/src/agresti.js
1169
1170 double n_tilde = n_total + binomial_z_sqr;
1171 double p_tilde = (n_hits + binomial_z_sqr_div_2)/n_tilde;
1172 double a = binomial_z*sqrt(p_tilde*(1 - p_tilde)/n_tilde);
1173
1174 return p_tilde + a;
1175 }

1188 inline double agresti_coull_lower_bound(uint64_t n_hits, uint64_t n_total) {
1189 // https://github.com/msn0/agresti-coull-interval/blob/master/src/agresti.js

1192 double p_tilde = (n_hits + binomial_z_sqr_div_2)/n_tilde;
1193 double a = binomial_z*sqrt(p_tilde*(1 - p_tilde)/n_tilde);

1222 inline double predict_time(CodeTreeNode &node) {
1223
1224 if (node.n_target <= 0)
1225 return PREDICT_MAX_TIME;
1226
1227 double lb = std::max(1e-4, agresti_coull_lower_bound(node.n_target, node.n_seen));
1228 double mu_hat = transform == tr_linear ? ((double) node.sum_time_d)/node.n_target : exp(((double) node.sum_time_d)/node.n_target);
1229
1230 return mu_hat/lb;
1231 }

1240 inline double predict_clip(Clip clip) {
1241
1242 int idx = 0, n = 0;
1243
1244 double t[MAX_SEQ_LEN_IN_PREDICT];
1245
1246 for (Clip::reverse_iterator it = clip.rbegin(); it != clip.rend(); it++) {
1247 ChildIndex::iterator jt = tree[idx].child.find(it->second);
1248
1249 if (jt == tree[idx].child.end())
1250 break;
1251
1252 idx = jt->second;
1253
1254 t[n++] = predict_time(tree[idx]);
1255 }
1256
1257 if (n == 0)
1258 return predict_time(tree[0]);
1259
1260 if (aggregate == ag_longest)
1261 return t[n - 1];
1262
1263 double ret = t[0];
1264
1265 if (aggregate == ag_mean) {
1266 for (int i = 1; i < n; i++)
1267 ret += t[i];
1268
1269 return ret/n;
1270 }
1271
1272 for (int i = 1; i < n; i++)
1273 ret = std::min(ret, t[i]);
1274
1275 return ret;
1276 }
